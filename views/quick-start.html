Quick Start

This page will take you through some basic examples you can implement in your community server.

Basic Implementation

C# 
using BattleBitAPI;
using BattleBitAPI.Common;
using BattleBitAPI.Common.Enums;
using BattleBitAPI.Server;
using System.Numerics;

class Program
{
    static void Main(string[] args)
    {
        var listener = new ServerListener<MyPlayer>();
        listener.OnGetPlayerStats += OnGetPlayerStats;
        listener.Start(29294); // Start server on port 29294
        
        Thread.Sleep(-1); // Make the thread sleep indefinitly
    }

    // Client will break without this
    private static async Task<PlayerStats> OnGetPlayerStats(ulong SteamID, PlayerStats Stats)
    {
        // The stats param is a new PlayerStats object
        // This will return a default PlayerStats object
        return Stats;
    }
}

class MyPlayer : Player
{
    // Used to add new properties and methods to the Player class
    // Eg.
}
